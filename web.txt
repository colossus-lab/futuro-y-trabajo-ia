    <script>
        // Configuración Chart.js
        Chart.defaults.font.family = "'Inter', sans-serif";
        Chart.defaults.color = '#64748b';

        // 1. Adoption Gap Chart
        const gapCtx = document.getElementById('adoptionGapChart').getContext('2d');
        new Chart(gapCtx, {
            type: 'bar',
            data: {
                labels: ['Capacidad Benchmark', 'Uso Global', 'Realidad Argentina'],
                datasets: [{
                    label: 'Nivel de Exposición (%)',
                    data: [100, 42, 28],
                    backgroundColor: ['#e2e8f0', '#94a3b8', '#74ACDF'],
                    borderRadius: 10,
                    barThickness: 35
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: { legend: { display: false } },
                scales: {
                    y: { beginAtZero: true, max: 100, ticks: { callback: v => v + '%' } },
                    x: { grid: { display: false } }
                }
            }
        });

        // 2. Radar Chart - Demographics
        const radarCtx = document.getElementById('demographicRadar').getContext('2d');
        const demoData = {
            general: [65, 55, 75, 80, 40],
            educacion: [95, 85, 30, 20, 90],
            genero: [45, 90, 85, 60, 55]
        };

        const demoChart = new Chart(radarCtx, {
            type: 'radar',
            data: {
                labels: ['Educación Superior', 'Ingresos Altos', 'Trabajo Cognitivo', 'Edad +45', 'Representación Femenina'],
                datasets: [{
                    label: 'Vulnerabilidad',
                    data: demoData.general,
                    backgroundColor: 'rgba(116, 172, 223, 0.2)',
                    borderColor: '#74ACDF',
                    pointBackgroundColor: '#F6B40E',
                    borderWidth: 3
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                scales: {
                    r: {
                        beginAtZero: true,
                        max: 100,
                        ticks: { display: false },
                        grid: { color: '#e2e8f0' }
                    }
                }
            }
        });

        function updateDemo(type, btn) {
            document.querySelectorAll('#demo-controls button').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
            demoChart.data.datasets[0].data = demoData[type];
            demoChart.update();
        }

        // 3. Sector Chart
        const sectorCtx = document.getElementById('sectorBarChart').getContext('2d');
        const sectorData = {
            share: [10, 15, 12, 45, 18],
            risk: [85, 60, 35, 15, 5]
        };
        
        const sectorChart = new Chart(sectorCtx, {
            type: 'bar',
            data: {
                labels: ['Software/Conocimiento', 'Sector Público', 'Industria', 'Comercio/Informal', 'Agro/Energía'],
                datasets: [{
                    label: 'Porcentaje (%)',
                    data: sectorData.share,
                    backgroundColor: '#74ACDF',
                    borderRadius: 5
                }]
            },
            options: {
                indexAxis: 'y',
                responsive: true,
                maintainAspectRatio: false,
                plugins: { legend: { display: false } },
                scales: {
                    x: { beginAtZero: true, max: 100 },
                    y: { grid: { display: false } }
                }
            }
        });

        function updateSector(mode) {
            const btns = event.currentTarget.parentElement.querySelectorAll('button');
            btns.forEach(b => {
                b.classList.remove('bg-white', 'shadow-sm');
                b.classList.add('text-slate-500');
            });
            event.currentTarget.classList.add('bg-white', 'shadow-sm');
            event.currentTarget.classList.remove('text-slate-500');
            
            sectorChart.data.datasets[0].data = sectorData[mode];
            sectorChart.update();
        }
    </script>
